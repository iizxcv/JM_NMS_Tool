## 백엔드 환경 구성 (Python)
    #1. 프로젝트 디렉토리로 이동
    cd /home/administrator/Service/nms-backend

    # 2. 기존 venv가 꼬였을 수 있으니 삭제 후 재생성
    rm -rf venv
    python3 -m venv venv

    # 3. 가상환경 활성화
    source venv/bin/activate

    # 4. uvicorn을 포함한 필수 패키지 설치
    # uvicorn[standard]를 설치해야 실행 파일이 정상적으로 생성됩니다.
    pip install --upgrade pip
    pip install "uvicorn[standard]" fastapi netmiko sqlalchemy psycopg2-binary pandas openpyxl

    2. 실행 파일 존재 여부 확인
    설치가 완료되었다면, 실제로 파일이 해당 경로에 있는지 확인합니다.

    Bash
    ls -l /home/administrator/Service/nms-backend/venv/bin/uvicorn

    MGMT Engine (nms-engine.service), MSA API (nms-api.service) 서비스 설정
    ```
        [Unit]
        Description=NMS Management Engine (Port 8000)
        After=network.target postgresql.service

        [Service]
        User=root
        WorkingDirectory=/root/nms-backend
        ExecStart=/root/nms-backend/venv/bin/uvicorn main_engine:app --host 0.0.0.0 --port 8000
        Restart=always

        [Install]
        WantedBy=multi-user.target
    ```
    서비스 등록 적용 - 등록 - 시작 - 점검
    ```
        systemctl daemon-reload
        systemctl enable nms-engine nms-api  # 부팅 시 자동 실행
        systemctl start nms-engine nms-api   # 즉시 실행
        systemctl status nms-engine          # 상태 확인
    ```
